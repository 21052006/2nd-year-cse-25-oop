#include<iostream>
using namespace std;

class complex {
    int real, imaginary;
    public:
        complex(int real, int imaginary = 0) {
            real = real, imaginary = imaginary;
            if (imaginary>-1)
                cout << real << " + " << imaginary << "i" << endl;
            else
                cout << real << " - " << imaginary*(-1) << "i" << endl;
        }

        ~complex() {
            cout << "Destructor is invoked" << endl;
        }
};

iii. Create a class which stores a sting and its length as data members. Include all the constructors. Include a member function to join two strings and display the concatenated string. 
#include<iostream>
using namespace std;

class _string {
    string s;
    int l;
    public:
        _string(string ss) {
            s = ss;
            l = ss.length();
            cout << "String: " <<  s << "\nLength: " << l << endl;
        }
        string merge(string s1, string s2) {
            string merged = s1+s2;
            cout << "String after merging: " << merged << endl;
            return merged;
        }
};

int main() {
    _string s("Piyush Somani");
    s.merge("piyush ", "somani");
    return 0;
}

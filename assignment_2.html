b) return the resultant object C3=C1.add(C2) 
#include<iostream>

class __distance__ {
    int feet, inches;
    public: 
        void input() {
            cout << "Feet: ";
            cin >> feet;
            cout << "Inches: ";
            cin >> inches;
            if (inches>12) {
                feet+=(inches/12);
                inches%=12;
            }
        }

        void display() {
            cout << feet << " ft " << inches << " inches\n";
        }

        void add(__distance__ a, __distance__ b) {
            feet = a.feet + b.feet;
            inches = a.inches + b.inches;
            if (inches>12) {
                feet+=(inches/12);
                inches%=12;
            }
        }
        
        __distance__ add(__distance__ a) {

                temp.feet+=(temp.inches/12);
                temp.inches%=12;
            }
            return temp;
        }
};

Create a class which stores id, name, age and basic salary of an employee. Input data for n number of employees. Calculate the gross salary of all the employees and display it along with all other details in a tabular form.
[
    Gross salary= Basic salary + DA + HRA,
    DA = 80% of Basic salary,
    HRA=10% of Basic salary
] 
#include <iostream>
using namespace std;

class employee {
    int id, age;
    string name;
    float b_salary, da, hra, gross;
    public:
        void input(void) {
            cout << "Enter name:";
            cin.ignore();
            getline(cin, name);
            cout << "Enter id:";
            cin >> id;
            cout << "Enter age:";
            cin >> age;
            cout << "Enter basic salary:";
            cin >> b_salary;
        }
        void calculate(void) { 
            da = (80 / 100.0) * b_salary;
            hra = (10 / 100.0) * b_salary;
            gross = b_salary + da + hra;
        }
        void output(void) {
            cout << "\tName: " << name << "\tID: " << id << "\tAge: " << age << "\tGross Salary: " << gross << endl;
        }
};

int main() {
    int n;
    cout << "Enter no. of employees: ";
    cin >> n;
    class employee e[n];
    for (int i = 0; i < n; i++) {
        e[i].input();
        e[i].calculate();
    }
    cout << "Displaying\n";
    for (int i = 0; i < n; i++) {
        e[i].output();
    }
    return 0;
}
Create a class which stores x and y coordinates of a point. Calculate distance between two given points and display it. 
#include<iostream>
#include<math.h>
using namespace std;

class point {
    public:
        int x, y;
};

float distance(point a, point b) {
    return sqrtf(pow(a.x-b.x, 2) + pow(a.y-b.y, 2));
}

int main() {
    point _1, _2;
    cout << "Enter coordinates of 1st point: ";
    cin >> _1.x >> _1.y;
    cout << "Enter coordinates of 2nd point: ";
    cin >> _2.x >> _2.y;
    cout << "Distance = ";
    cout << distance(_1, _2);
    return 0;
}

#include<bits/stdc++.h>
using namespace std;

float circle(float radius) {
    int area, perimeter;
    area = 3.14*radius*radius;
    perimeter = 2*3.14*radius;
    cout << "\n\nArea: " << area << "\nPerimeter: " << perimeter << endl;
}

float rectangle(int length, int breadth) {
    int area, perimeter;
    area = length*breadth;
    perimeter = 2*(length + breadth);
    cout << "\n\nArea: " << area << "\nPerimeter: " << perimeter << endl;
}

float square(int length) {
    int area, perimeter;
    area = length*length;
    perimeter = 4*length;
    cout << "\n\nArea: " << area << "\nPerimeter: " << perimeter << endl;
}

int main() {
    int z = 0;
    while (1) {
        cout << "\n\npress 1 for circle\npress 2 for rectangle\npress 3 for square\nany other key to exit\nYour input: ";
        cin >> z;
        if (z == 1) {
            int radius;
            cout << "Enter the radius of circle: ";
            cin >> radius;
            circle(radius);
        } else if (z == 2) {
            int length, breadth;
            cout << "Entre the length and breadth of the rectangle: ";
            cin >> length >> breadth;
            rectangle(length, breadth);
        } else if (z == 3) {
            int length;
            cout << "Entre the length of square: ";
            cin >> length;
            square(length);
        } else {
            cout << "Exit Successfull";
            break;
        }
    }
    
    return 0;
}

    cin >> a >> b >> c;
    d = sqrt((b*b) - (4*a*c));
    if (d == 0) {
        r1 = r2 = (-b - d)/(2*a);
        cout << "root1 = root2 = " << r1 << endl;
    } else if (d>0) {
        r1 = (-b-d)/(2*a);
        r2 = (-b+d)/(2*a);
        cout << "root1 = " << r1 << "\nroot2 = " << r2;
    } else {
        cout << "Complex roots" << endl;
    }

}

    int arr[5], used[5] = {-1, -1, -1, -1, -1};
    cout << "Enter the elements of the arr[5]: ";
    cin >> arr[0] >> arr[1] >> arr[2] >> arr[3] >> arr[4];
    for (int i = 0;i<5;i++) {
        int c = 0, flag = 1;
        for (int j = 0;j<5;j++) {
            if (used[j] == arr[i]) {
                flag = 0;
                break;
            }
        }
        if (flag) {
            used[i] = arr[i];
            for (int j = i;j<5;j++) {
                if (arr[j] == arr[i]) {
                    c++;
                }
            }
            cout << arr[i] << " occurs " << c << " times " << endl;
        }
    }
    return 0;
}
